    <script>
        new Vue({

                    accountDialogVisible: false,
                    accountTabActive: 'avatar',
                    previewAvatar: '',
                    usernameForm: {
                        newUsername: ''
                    },
                    usernameRules: {
                        newUsername: [
                            { required: true, message: '请输入新用户名', trigger: 'blur' },
                            { min: 2, max: 20, message: '用户名长度在2-20个字符', trigger: 'blur' }
                        ]
                    },
                    loginForm: {
                        username: '',
                        password: ''
                    },
                    loginRules: {
                        username: [
                            { required: true, message: '请输入管理员账号', trigger: 'blur' }
                        ],
                        password: [
                            { required: true, message: '请输入密码', trigger: 'blur' }
                        ]
                    },
                    signForm: {

                        name: '',
                        department: '',

                    },
                    rules: {
                        type: [

                        ],
                        name: [

                            { min: 2, max: 10, message: '长度在 2 到 10 个字符', trigger: 'blur' }
                        ],
                        department: [
                            { required: true, message: '请选择部门', trigger: 'change' }
                        ],
                        shift: [
                            { required: true, message: '请选择班次', trigger: 'change' }
                        ]
                    },
                    records: [],
                    filteredRecords: [],
                    filterDepartment: '',
                    filterShift: '',
                    filterDate: '',
                    todayStats: {
                        total: 0,
                        onDuty: 0,
                        completed: 0
                    },
                    // 管理员账号配置（实际使用时应该从后端获取）
                    adminAccounts: [
                        { username: 'admin', password: '123456', name: '系统管理员', avatar: '' },
                        { username: 'manager', password: '888888', name: '考勤管理员', avatar: '' }
                    ],
                    changePwdForm: {
                        oldPassword: '',
                        newPassword: '',
                        confirmPassword: ''
                    },
                    changePwdRules: {
                        oldPassword: [
                            { required: true, message: '请输入当前密码', trigger: 'blur' }
                        ],
                        newPassword: [
                            { required: true, message: '请输入新密码', trigger: 'blur' },
                            { min: 6, message: '密码长度至少6位', trigger: 'blur' }
                        ],
                        confirmPassword: [
                            { required: true, message: '请再次输入新密码', trigger: 'blur' },
                            { validator: (rule, value, callback) => {
                                if (value !== this.changePwdForm.newPassword) {
                                    callback(new Error('两次输入的密码不一致'));
                                } else {
                                    callback();
                                }
                            }, trigger: 'blur' }
                        ]
                    }
                };
            },
            mounted() {
                this.loadRecords();
                this.calculateStats();
                this.checkLoginStatus();
                this.loadAdminAccounts(); // 加载管理员账号
            },
            methods: {
                // 加载管理员账号（从localStorage读取，如果有自定义的话）
                loadAdminAccounts() {
                    const savedAccounts = localStorage.getItem('adminAccounts');
                    if (savedAccounts) {
                        this.adminAccounts = JSON.parse(savedAccounts);
                    }
                },

                // 保存管理员账号到localStorage
                saveAdminAccounts() {
                    localStorage.setItem('adminAccounts', JSON.stringify(this.adminAccounts));
                },

                // 显示账户设置对话框
                showAccountDialog() {
                    this.accountDialogVisible = true;
                    this.accountTabActive = 'avatar';
                    this.previewAvatar = this.currentAdminAvatar;
                    this.usernameForm.newUsername = '';
                    this.changePwdForm = {
                        oldPassword: '',
                        newPassword: '',
                        confirmPassword: ''
                    };
                },

                // 处理头像上传
                handleAvatarChange(event) {
                    const file = event.target.files[0];
                    if (!file) return;

                    // 检查文件类型
                    if (!file.type.match('image.*')) {
                        this.$message.error('请选择图片文件！');
                        return;
                    }

                    // 检查文件大小（限制2MB）
                    if (file.size > 2 * 1024 * 1024) {
                        this.$message.error('图片大小不能超过2MB！');
                        return;
                    }

                    // 读取文件并预览
                    const reader = new FileReader();
                    reader.onload = (e) => {
                        this.previewAvatar = e.target.result;
                        this.currentAdminAvatar = e.target.result;
                        
                        // 保存到管理员账户
                        const savedAdmin = localStorage.getItem('adminLoggedIn');
                        if (savedAdmin) {
                            const adminData = JSON.parse(savedAdmin);
                            const adminIndex = this.adminAccounts.findIndex(
                                a => a.username === adminData.username
                            );
                            if (adminIndex !== -1) {
                                this.adminAccounts[adminIndex].avatar = e.target.result;
                                this.saveAdminAccounts();
                                
                                // 更新登录状态
                                adminData.avatar = e.target.result;
                                localStorage.setItem('adminLoggedIn', JSON.stringify(adminData));
                                
                                this.$message.success('头像上传成功！');
                            }
                        }
                    };
                    reader.readAsDataURL(file);
                    
                    // 清空input，允许重复上传同一文件
                    event.target.value = '';
                },

                // 提交修改用户名
                submitChangeUsername() {
                    this.$refs.usernameForm.validate((valid) => {
                        if (valid) {
                            const savedAdmin = localStorage.getItem('adminLoggedIn');
                            if (!savedAdmin) {
                                this.$message.error('登录状态异常，请重新登录');
                                return;
                            }
                            
                            const adminData = JSON.parse(savedAdmin);
                            const adminIndex = this.adminAccounts.findIndex(
                                a => a.username === adminData.username
                            );
                            
                            if (adminIndex === -1) {
                                this.$message.error('管理员账号不存在');
                                return;
                            }
                            
                            // 修改用户名
                            this.adminAccounts[adminIndex].name = this.usernameForm.newUsername;
                            this.currentAdmin = this.usernameForm.newUsername;
                            
                            // 保存到localStorage
                            this.saveAdminAccounts();
                            
                            // 更新登录状态
                            adminData.name = this.usernameForm.newUsername;
                            localStorage.setItem('adminLoggedIn', JSON.stringify(adminData));
                            
                            this.$message.success('用户名修改成功！');
                            this.usernameForm.newUsername = '';
                            this.$refs.usernameForm.resetFields();
                        } else {
                            this.$message.error('请正确填写用户名');
                            return false;
                        }
                    });
                },

                // 取消修改密码
                cancelChangePwd() {
                    this.accountDialogVisible = false;
                    if (this.$refs.changePwdForm) {
                        this.$refs.changePwdForm.resetFields();
                    }
                },

                // 提交修改密码
                submitChangePwd() {
                    this.$refs.changePwdForm.validate((valid) => {
                        if (valid) {
                            // 获取当前登录的管理员信息
                            const savedAdmin = localStorage.getItem('adminLoggedIn');
                            if (!savedAdmin) {
                                this.$message.error('登录状态异常，请重新登录');
                                return;
                            }
                            
                            const adminData = JSON.parse(savedAdmin);
                            const currentUsername = adminData.username;
                            
                            // 查找当前管理员账号
                            const adminIndex = this.adminAccounts.findIndex(
                                a => a.username === currentUsername
                            );
                            
                            if (adminIndex === -1) {
                                this.$message.error('管理员账号不存在');
                                return;
                            }
                            
                            // 验证旧密码
                            if (this.adminAccounts[adminIndex].password !== this.changePwdForm.oldPassword) {
                                this.$message.error('当前密码错误！');
                                return;
                            }
                            
                            // 修改密码
                            this.adminAccounts[adminIndex].password = this.changePwdForm.newPassword;
                            
                            // 保存到localStorage
                            this.saveAdminAccounts();
                            
                            this.$message.success('密码修改成功！');
                            this.changePwdForm = {
                                oldPassword: '',
                                newPassword: '',
                                confirmPassword: ''
                            };
                            this.$refs.changePwdForm.resetFields();
                        } else {
                            this.$message.error('请正确填写所有信息');
                            return false;
                        }
                    });
                },

                // 检查登录状态
                checkLoginStatus() {
                    const savedAdmin = localStorage.getItem('adminLoggedIn');
                    if (savedAdmin) {
                        const adminData = JSON.parse(savedAdmin);
                        this.isAdminLoggedIn = true;
                        this.currentAdmin = adminData.name;
                        this.currentUsername = adminData.username;
                        this.currentAdminAvatar = adminData.avatar || this.currentAdminAvatar;
                        this.previewAvatar = this.currentAdminAvatar;
                    }
                },

                // 处理标签页切换
                handleTabClick(tab) {
                    if (tab.name === 'view' && !this.isAdminLoggedIn) {
                        this.$message.warning('请先登录管理员账号');
                    }
                },

                // 管理员登录
                handleLogin() {
                    this.$refs.loginForm.validate((valid) => {
                        if (valid) {
                            // 从最新的adminAccounts中查找（可能已被修改）
                            const admin = this.adminAccounts.find(
                                a => a.username === this.loginForm.username && 
                                     a.password === this.loginForm.password
                            );

                            if (admin) {
                                this.isAdminLoggedIn = true;
                                this.currentAdmin = admin.name;
                                this.currentUsername = admin.username;
                                this.currentAdminAvatar = admin.avatar || this.currentAdminAvatar;
                                this.previewAvatar = this.currentAdminAvatar;
                                
                                // 保存登录状态
                                localStorage.setItem('adminLoggedIn', JSON.stringify({
                                    username: admin.username,
                                    name: admin.name,
                                    avatar: admin.avatar || this.currentAdminAvatar
                                }));

                                this.$message.success('登录成功！欢迎回来，' + admin.name);
                                this.loginForm.username = '';
                                this.loginForm.password = '';
                                
                                // 登录成功后加载数据
                                this.loadRecords();
                                this.calculateStats();
                            } else {
                                this.$message.error('账号或密码错误！');
                            }
                        }
                    });
                },

                // 退出登录
                handleLogout() {
                    this.$confirm('确定要退出登录吗？', '提示', {
                        confirmButtonText: '确定',
                        cancelButtonText: '取消',
                        type: 'warning'
                    }).then(() => {
                        this.isAdminLoggedIn = false;
                        this.currentAdmin = '';
                        localStorage.removeItem('adminLoggedIn');
                        this.$message.success('已退出登录');
                        this.activeTab = 'sign';
                    }).catch(() => {});
                },

                // 提交签到
                submitSign() {
                    this.$refs.signForm.validate((valid) => {
                        if (valid) {
                            const currentTime = this.getCurrentTime();
                            const today = this.getToday();
                            
                            // 查找是否已有该人员今日该班次的记录
                            let existingRecord = this.records.find(r => 
                                r.name === this.signForm.name && 
                                r.shift === this.signForm.shift && 
                                r.department === this.signForm.department &&
                                r.date === today
                            );

                            if (this.signForm.type === '进站') {
                                if (existingRecord && existingRecord.checkInTime) {
                                    this.$message.warning('您今日该班次已经进站，无需重复签到！');
                                    return;
                                }
                                
                                if (existingRecord) {
                                    // 更新进站时间
                                    existingRecord.checkInTime = currentTime;
                                    existingRecord.remark = this.signForm.remark;
                                } else {
                                    // 创建新记录
                                    const newRecord = {
                                        id: this.records.length + 1,
                                        name: this.signForm.name,
                                        department: this.signForm.department,
                                        shift: this.signForm.shift,
                                        checkInTime: currentTime,
                                        checkOutTime: '',
                                        remark: this.signForm.remark,
                                        date: today
                                    };
                                    this.records.unshift(newRecord);
                                }
                                this.$message.success(`${this.signForm.name} 进站签到成功！`);
                            } else {
                                // 出站
                                if (!existingRecord || !existingRecord.checkInTime) {
                                    this.$message.error('请先进站签到！');
                                    return;
                                }
                                if (existingRecord.checkOutTime) {
                                    this.$message.warning('您今日该班次已经出站，无需重复签到！');
                                    return;
                                }
                                existingRecord.checkOutTime = currentTime;
                                if (this.signForm.remark) {
                                    existingRecord.remark += ' | ' + this.signForm.remark;
                                }
                                this.$message.success(`${this.signForm.name} 出站签到成功！`);
                            }

                            this.saveRecords();
                            this.filterRecords();
                            this.calculateStats();
                            this.resetForm();
                        } else {
                            this.$message.error('请完整填写表单信息！');
                            return false;
                        }
                    });
                },

                // 重置表单
                resetForm() {
                    this.$refs.signForm.resetFields();
                    this.signForm.type = '进站';
                },

                // 加载记录
                loadRecords() {
                    const savedRecords = localStorage.getItem('attendanceRecords');
                    if (savedRecords) {
                        this.records = JSON.parse(savedRecords);
                    }
                    this.filterRecords();
                    this.calculateStats();
                    this.$message.success('数据已刷新');
                },

                // 保存记录
                saveRecords() {
                    localStorage.setItem('attendanceRecords', JSON.stringify(this.records));
                },

                // 删除记录
                deleteRecord(id) {
                    this.$confirm('确定要删除这条记录吗？', '提示', {
                        confirmButtonText: '确定',
                        cancelButtonText: '取消',
                        type: 'warning'
                    }).then(() => {
                        this.records = this.records.filter(r => r.id !== id);
                        this.saveRecords();
                        this.filterRecords();
                        this.calculateStats();
                        this.$message.success('删除成功！');
                    }).catch(() => {
                        this.$message.info('已取消删除');
                    });
                },

                // 清空所有记录
                clearAllRecords() {
                    this.$confirm('确定要清空所有记录吗？此操作不可恢复！', '警告', {
                        confirmButtonText: '确定',
                        cancelButtonText: '取消',
                        type: 'error'
                    }).then(() => {
                        this.records = [];
                        this.filteredRecords = [];
                        localStorage.removeItem('attendanceRecords');
                        this.calculateStats();
                        this.$message.success('所有记录已清空！');
                    }).catch(() => {
                        this.$message.info('已取消操作');
                    });
                },

                // 筛选记录
                filterRecords() {
                    this.filteredRecords = this.records.filter(record => {
                        let match = true;
                        if (this.filterDepartment && record.department !== this.filterDepartment) {
                            match = false;
                        }
                        if (this.filterShift && record.shift !== this.filterShift) {
                            match = false;
                        }
                        if (this.filterDate && record.date !== this.filterDate) {
                            match = false;
                        }
                        return match;
                    });
                },

                // 计算统计数据
                calculateStats() {
                    const today = this.getToday();
                    const todayRecords = this.records.filter(r => r.date === today);
                    
                    this.todayStats.total = todayRecords.length;
                    this.todayStats.onDuty = todayRecords.filter(r => r.checkInTime && !r.checkOutTime).length;
                    this.todayStats.completed = todayRecords.filter(r => r.checkInTime && r.checkOutTime).length;
                },

                // 导出数据
                exportData() {
                    if (this.records.length === 0) {
                        this.$message.warning('暂无数据可导出');
                        return;
                    }

                    let csvContent = '\uFEFF'; // UTF-8 BOM
                    csvContent += '序号,部门,姓名,班次,进站时间,出站时间,备注,日期\n';
                    
                    this.filteredRecords.forEach(record => {
                        csvContent += `${record.id},${record.department},${record.name},${record.shift},`;
                        csvContent += `${record.checkInTime || '未进站'},${record.checkOutTime || '未出站'},`;
                        csvContent += `${record.remark || '-'},${record.date}\n`;
                    });

                    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
                    const link = document.createElement('a');
                    const url = URL.createObjectURL(blob);
                    link.setAttribute('href', url);
                    link.setAttribute('download', `考勤记录_${this.getToday()}.csv`);
                    link.style.visibility = 'hidden';
                    document.body.appendChild(link);
                    link.click();
                    document.body.removeChild(link);
                    
                    this.$message.success('数据导出成功！');
                },

                // 获取当前时间
                getCurrentTime() {
                    const now = new Date();
                    const year = now.getFullYear();
                    const month = String(now.getMonth() + 1).padStart(2, '0');
                    const day = String(now.getDate()).padStart(2, '0');
                    const hours = String(now.getHours()).padStart(2, '0');
                    const minutes = String(now.getMinutes()).padStart(2, '0');
                    const seconds = String(now.getSeconds()).padStart(2, '0');
                    return `${year}-${month}-${day} ${hours}:${minutes}:${seconds}`;
                },

                // 获取今日日期
                getToday() {
                    const now = new Date();
                    const year = now.getFullYear();
                    const month = String(now.getMonth() + 1).padStart(2, '0');
                    const day = String(now.getDate()).padStart(2, '0');
                    return `${year}-${month}-${day}`;
                }
            }
        });
